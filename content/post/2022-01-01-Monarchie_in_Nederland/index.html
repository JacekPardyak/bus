The question

How to present a list of Dutch monarchs and their parents as a diagram?

library(magrittr)
library(DiagrammeR)
library(DiagrammeRsvg)

'Monarchie_in_Nederland.gv' %>%
  grViz() %>%
  export_svg() %>%
  writeLines('Monarchie_in_Nederland.svg')
